## 题目

按之字形顺序打印二叉树

请实现一个函数按照之字形打印二叉树，即第一行按照从左到右的顺序打印，第二层按照从右至左的顺序打印，第三行按照从左到右的顺序打印，其他行以此类推。

## JS实现

```
/* function TreeNode(x) {

function Print(pRoot)
{
    // write code here
    let result = [];

    if(pRoot === null) { return result; }

    let q = [];
    q.push(pRoot);

    let rev = true;

    while(q.length != 0){
        let level = [];
        let len = q.length;
        for(let i = 0; i < len; i++){
            const node = q.shift();
            if(rev){
                level.push(node.val);
            }else {
                level.unshift(node.val);
            }
            if(node.left != null) { q.push(node.left); }
            if(node.right != null) { q.push(node.right); }
        }
        result.push(level);
        rev=!rev;
    }
    return result;
}
```
